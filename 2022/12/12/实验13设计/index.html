表设计

图书信息表book_list

属性名 说明 数据类型 约束
book_id 书号 int(10) PRIMARY KEY
name 书名 char(255) PRIMARY KEY
press 出版社 char(255) NOT NULL
price 价格 money NOT NULL
keywords 关键字 char(255) NOT NULL
calatog 目录 char(255)
cover 封皮 char(255)

作者信息表author_list

属性名 说明 数据类型 约束
author 作者 char(255) PRIMARY KEY
book_id 书号 int(10) PRIMARY KEY, FOREIGN KEY book_list(book_id)
order 顺序号 int(1) 1<= order <= 4

关键字表keywords_list

属性名 说明 数据类型 约束
keywords 关键字 char(255) PRIMARY KEY
book_id 书号 int(10) PRIMARY KEY, FOREIGN KEY book_list(book_id)

供应表provide_list

属性名 说明 数据类型 约束
book_id 书号 int(10) PRIMARY KEY, FOREIGN KEY book_list(book_id)
provider_id 供应商号 int(10) PRIMARY KEY, FOREIGN KEY provider_list(provider_id)

库存情况表reserve_list

属性名 说明 数据类型 约束
book_id 书号 int(10) PRIMARY KEY, FOREIGN KEY book_list(book_id)
num 存货量 int(10) NOT NULL
position 存放位置 char(255)

缺书记录表lack_list

属性名 说明 数据类型 约束
book_id 书号 int(10) PRIMARY KEY, FOREIGN KEY book_list(book_id)
name 书名 char(255) NOT NULL
press 出版社 char(255) NOT NULL
provider 供应商 char(255) NOT NULL
num 数量 int(10) NOT NULL
register_date 登记日期 date NOT NULL

采购单表purchase_list

属性名 说明 数据类型 约束
book_id 书号 int(10) PRIMARY KEY, FOREIGN KEY book_list(book_id)
num 数量 int(10) NOT NULL
email 顾客email char(255)

客户信息表customer_list

属性名 说明 数据类型 约束
customer_id 客户ID int(10) PRIMARY KEY
password 登陆密码 char(255) NOT NULL
name 客户名称 char(255) NOT NULL
address 客户地址 char(255) NOT NULL
balance 账户余额 char(255) NOT NULL
credit 信用等级 int(1)
credit_limit 信用额度 int(10)

订单表order_list

属性名 说明 数据类型 约束
order_id 订单号 int(10) PRIMARY KRY
customer_id 客户ID int(10) FOREIGN KEY customer_list(customer_id)
order_date 订货日期 date NOT NULL
order_address 发货地址 char(255) NOT NULL
order_pay 订单金额 money NOT NULL
delivery_state 发货情况 char(255) NOT NULL

订书信息表ordered_books_list

属性名 说明 数据类型 约束
order_id 订单号 int(10) PRIMARY KEY, FOREIGN KEY order_list(order_id)
book_id 书号 int(10) PRIMARY KEY, FOREIGN KEY book_list(book_id)
num 订书数量 int(10) NOT NULL

供应商provider_list

属性名 说明 数据类型 约束
provider_id 供应商编号 int(10) PRIMARY KEY
